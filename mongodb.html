  salary:60000 }

{ name:"Priya", salary:55000 }

Q25. limit() — Top 3 Salaries
SQL: SELECT * FROM employees ORDER BY salary DESC LIMIT 3;
💻 Command
db.employees.find(
  {},
  { name:1, salary:1, _id:0 }

 .limit(3)

 .skip(2)
 .limit(1)

{ name:"Priya", salary:55000 }
(1 document)

// skip(0)→Suman 70k
// skip(1)→Amit  60k
// skip(2)→Priya 55k ✓

Section 10 — Aggregation Pipeline 🔥

🔥 Aggregation = MongoDB's most powerful feature. It's like SQL's GROUP BY + JOIN + subqueries all in one pipeline. Stages flow: $match → $group → $sort → $project → $lookup → $limit
Q44. $group — Department-wise Count
SQL: SELECT department, COUNT(*) FROM employees GROUP BY department;
💻 Command
db.employees.aggregate([
  {
    $group: {
      _id: "$department",
      count: { $sum: 1 }
    }
  }
])
📋 Output
{ _id:"IT",    count:3 }
{ _id:"HR",    count:2 }
{ _id:"Sales", count:1 }
Q45. $group — Avg, Max, Min, Sum
💻 Command
db.employees.aggregate([
  {
    $group: {
      _id: "$department",
      avgSal: { $avg: "$salary" },
      maxSal: { $max: "$salary" },
      minSal: { $min: "$salary" },
      totalSal: { $sum: "$salary"}
    }
  }
])
📋 Output
{ _id: "IT",
  avgSal: 61666.67,
  maxSal: 70000,
  minSal: 55000,
  totalSal: 185000 }
{ _id: "HR",
  avgSal: 42500,
  maxSal: 45000,
  minSal: 40000,
  totalSal: 85000 }
{ _id: "Sales",
  avgSal: 35000,
  maxSal: 35000,
  minSal: 35000,
  totalSal: 35000 }
Q46. $match + $group — Filter then Group
SQL: SELECT department, AVG(salary) FROM employees WHERE salary > 40000 GROUP BY department;
💻 Command
db.employees.aggregate([
  { $match: {
      salary: { $gt: 40000 }
  }},
  { $group: {
      _id: "$department",
      avgSal: { $avg: "$salary" }
  }}
])
📋 Output
{ _id: "IT",  avgSal: 61666.67 }
{ _id: "HR",  avgSal: 45000 }
(2 results)

// Neha(35k) & Ravi(40k)
// filtered out by $match
Q47. $group + $match — HAVING Equivalent
SQL: SELECT dept, COUNT(*) FROM employees GROUP BY dept HAVING COUNT(*) > 1;
💻 Command
db.employees.aggregate([
  { $group: {
      _id: "$department",
      count: { $sum: 1 }
  }},
  { $match: {
      count: { $gt: 1 }
  }}
])
📋 Output
{ _id: "IT", count: 3 }
{ _id: "HR", count: 2 }
(2 results)

// Sales has only 1 employee
// filtered out by HAVING
Q48. $sort in Aggregation
💻 Command
db.employees.aggregate([

      _id: "$department",

📋 Output
{ empName:"AMIT",
  department:"IT",
  annualSalary:720000 }
{ empName:"RAVI",
  department:"HR",
  annualSalary:480000 }

Q50. $lookup — JOIN Equivalent 🔥
SQL: SELECT * FROM employees e JOIN departments d ON e.department = d.dept_name;
💻 Command
db.employees.aggregate([
  {
    $lookup: {
      from: "departments",
      localField: "department",
      foreignField: "dept_name",
      as: "dept_info"
    }
  }
])
📋 Output
{ _id:1, name:"Amit",
  department:"IT",
  salary:60000,
  dept_info: [{
    _id:1,
    dept_name:"IT",
    location:"Bangalore"
  }]
}
... (each doc gets
     dept_info array)
Q51. $unwind — Flatten Arrays
$unwind creates a separate document for each array element. Essential after $lookup or for array analysis.
💻 Command
db.employees.aggregate([
  { $match: { name: "Suman" } },
  { $unwind: "$skills" }
])
📋 Output
{ _id:6, name:"Suman",
  skills:"React" }
{ _id:6, name:"Suman",
  skills:"Node" }
{ _id:6, name:"Suman",
  skills:"AWS" }
(3 documents from 1!)
Q52. $lookup + $unwind — Clean Join
$lookup returns array. $unwind flattens it to get clean joined result.
💻 Command
db.employees.aggregate([
  { $lookup: {
      from: "departments",
      localField: "department",
      foreignField: "dept_name",
      as: "dept"
  }},
  { $unwind: "$dept" },
  { $project: {
      name: 1,
      department: 1,
      location: "$dept.location",
      _id: 0
  }}
])
📋 Output
{ name:"Amit",
  department:"IT",
  location:"Bangalore" }
{ name:"Ravi",
  department:"HR",
  location:"Mumbai" }
{ name:"Priya",
  department:"IT",
  location:"Bangalore" }
{ name:"Neha",
  department:"Sales",
  location:"Delhi" }
{ name:"Arjun",
  department:"HR",
  location:"Mumbai" }
{ name:"Suman",
  department:"IT",
  location:"Bangalore" }
(6 documents — clean join!)

Section 11 — Indexing (Performance)

💡 Indexes speed up read queries but slow down writes. By default, MongoDB creates an index on _id. Always index fields you frequently filter/sort on.
Q53. createIndex() — Single Field Index
💻 Command
db.employees.createIndex(
  { salary: 1 }
)

// 1 = ascending index
// -1 = descending index
📋 Output
{
  "createdCollectionAutomatically": false,
  "numIndexesBefore": 1,

Q66. Complete REST API Routes
This is the typical MERN backend pattern — asked in every MERN interview.
💻 Code (routes/employee.js)
const express = require('express');
const router = express.Router();
const Employee =
  require('../models/Employee');

// GET all
router.get('/',
  async (req, res) => {
  const emps =
    await Employee.find();
  res.json(emps);
});

// GET by id
router.get('/:id',
  async (req, res) => {
  const emp =
    await Employee.findById(
      req.params.id);
  res.json(emp);
});

// POST create
router.post('/',
  async (req, res) => {
  const emp =
    await Employee.create(
      req.body);
  res.status(201).json(emp);
});

// PUT update
router.put('/:id',
  async (req, res) => {
  const emp =
    await Employee
      .findByIdAndUpdate(
        req.params.id,
        req.body,
        { new: true }
      );
  res.json(emp);
});

// DELETE
router.delete('/:id',
  async (req, res) => {
  await Employee
    .findByIdAndDelete(
      req.params.id);
  res.json({ msg: "Deleted" });
});

module.exports = router;

→ Create new employee
  Body: { name, dept, salary }

PUT    /api/employees/:id
→ Update employee
  Body: { salary: 70000 }

DELETE /api/employees/:id
→ Delete employee
