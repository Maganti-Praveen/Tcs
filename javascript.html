5. Scope & Hoisting
Scope determines variable accessibility. Hoisting moves declarations to the top (conceptually).
💻 Code
// === SCOPE ===
// 1. Global scope
var globalVar = "I'm global";

// 2. Function scope
function myFunc() {
    var funcVar = "I'm function-scoped";
    console.log(globalVar);  // ✅
    console.log(funcVar);    // ✅
}
// console.log(funcVar); // ❌ ReferenceError

// 3. Block scope (let/const)
if (true) {
    let blockLet = "block";
    var blockVar = "function";
}
console.log(blockVar);   // ✅ "function"
// console.log(blockLet); // ❌ ReferenceError

// 4. Lexical scope (closure)
function outer() {
    let outerVar = "outer";
    function inner() {
        console.log(outerVar); // ✅ Access parent
    }
    inner();
}
outer();

// === HOISTING ===
// var is hoisted (initialized as undefined)
console.log(x); // undefined
var x = 5;

// function declarations are fully hoisted
sayHi(); // ✅ "Hi!"
function sayHi() { console.log("Hi!"); }

// let/const are hoisted but in TDZ
// console.log(y); // ❌ ReferenceError
let y = 10;

// function expressions NOT hoisted
// sayBye(); // ❌ TypeError
const sayBye = function() { console.log("Bye!"); };

// Fill
[1,2,3].fill(0);     // [0,0,0]
[1,2,3].fill(9,1,2); // [1,9,3]

1. Closures — Deep Dive
A closure is a function that remembers its outer scope even after the outer function has returned.
💻 Code
// Basic closure
function outer() {
    let count = 0;
    return function inner() {
        count++;
        return count;
    };
}
const counter = outer();
console.log(counter()); // 1
console.log(counter()); // 2
console.log(counter()); // 3

// Practical: Private variables
function createBankAccount(initial) {
    let balance = initial; // private!
    return {
        deposit(amt) { balance += amt; },
        withdraw(amt) {
            if (amt > balance) return "Insufficient";
            balance -= amt;
        },
        getBalance() { return balance; }
    };
}
const acc = createBankAccount(1000);
acc.deposit(500);
acc.withdraw(200);
console.log(acc.getBalance()); // 1300
// acc.balance → undefined (private!)

2. 'this' Keyword — All Rules
this depends on how a function is called, not where it's defined.
💻 Code
// 1. Global context
console.log(this); // window (browser) / global (node)

// 2. Object method → this = object
const obj = {
    name: "Alice",
    greet() {
        console.log(this.name); // "Alice"
    }
};
obj.greet();

// 3. Regular function → this = window/undefined(strict)
function show() {
    console.log(this); // window (or undefined in strict)
}

// 4. Arrow function → this = parent scope (lexical)
const obj2 = {
    name: "Bob",
    greet: () => {
        console.log(this.name); // undefined! (inherits outer)
    },
    delayedGreet() {
        setTimeout(() => {
            console.log(this.name); // "Bob" ✅ (arrow inherits)
        }, 100);
    }
};

// 5. Constructor → this = new object
function Person(name) {
    this.name = name;
}
const p = new Person("Charlie");
console.log(p.name); // "Charlie"

// 6. Explicit binding
function sayHi() {
    console.log(`Hi, ${this.name}`);
}

const user = { name: "Dave" };

// call — invoke immediately with args
sayHi.call(user);         // "Hi, Dave"

4. Debounce & Throttle
Debounce: Wait until user stops. Throttle: Execute at most once per interval.
💻 Debounce
// Debounce: Waits N ms after LAST call
// Use: search input, window resize
function debounce(fn, delay) {
    let timer;
    return function(...args) {
        clearTimeout(timer);
        timer = setTimeout(() => {
            fn.apply(this, args);
        }, delay);
    };
}

// Usage
const search = debounce((query) => {
    console.log("Searching:", query);
}, 300);

// Type: h → he → hel → hello
// Only fires ONCE: "Searching: hello"
// (after 300ms of no typing)
💻 Throttle
// Throttle: At most once per N ms
// Use: scroll events, button clicks
function throttle(fn, limit) {
    let inThrottle = false;
    return function(...args) {
        if (!inThrottle) {
            fn.apply(this, args);
            inThrottle = true;
            setTimeout(() => {
                inThrottle = false;
            }, limit);
        }
    };
}

// Usage
const onScroll = throttle(() => {
    console.log("Scroll event!");
}, 200);

window.addEventListener("scroll", onScroll);
// Fires at most every 200ms during scroll

// Comparison:
// Debounce → fires AFTER silence
// Throttle → fires DURING action (limited)

// Currying: f(a,b,c) → f(a)(b)(c)
function curry(fn) {
    return function curried(...args) {
        if (args.length >= fn.length) {
            return fn.apply(this, args);
        }
        return function(...args2) {
            return curried.apply(this, [...args, ...args2]);
        };
    };
}

6. Shallow Copy vs Deep Copy
Shallow copy copies one level. Deep copy copies all nested levels.
💻 Code

    hobbies: ["reading", "coding"]

// Method 1: JSON (simple but limited)

// ⚠️ Fails with: functions, undefined, Date, RegExp,
//    Map, Set, Infinity, NaN, circular references

// Method 2: structuredClone (modern, recommended!)
const deep2 = structuredClone(original);
// ✅ Handles: Date, Map, Set, ArrayBuffer, circular refs
// ❌ Fails with: functions, DOM nodes, symbols

deep2.address.city = "Kolkata";
console.log(original.address.city); // "Delhi" (safe!)

// Method 3: Recursive function
function deepClone(obj) {
    if (obj === null || typeof obj !== "object") return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof Array) return obj.map(deepClone);
    const clone = {};
    for (const key in obj) {
        if (obj.hasOwnProperty(key)) {
            clone[key] = deepClone(obj[key]);
        }
    }
    return clone;
}

        (this.events[event] || []).forEach(fn => fn(...args));
    }

    off(event, fn) {
        this.events[event] = (this.events[event] || [])
            .filter(f => f !== fn);
    }
}

JSHard Q52. Implement deep clone.
// Best: structuredClone(obj) — native, modern

// Manual recursive:
function deepClone(obj) {
    if (obj === null || typeof obj !== "object") return obj;
    if (obj instanceof Date) return new Date(obj);
    if (obj instanceof Array) return obj.map(deepClone);
    const clone = {};
    for (const key in obj) {
        if (obj.hasOwnProperty(key))
            clone[key] = deepClone(obj[key]);
    }
    return clone;
}
