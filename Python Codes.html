Basic Problems

1. Fibonacci Series
Generate the first n Fibonacci numbers using iteration.
def fibo(n):
    r=[]
    a=0
    b=1
    for i in range(n):
        r.append(a)
        a,b=b,a+b
    return r
2. Factorial (Recursion)
Calculate factorial of n using recursion. Returns "Not Possible" for negative numbers.
def fact(n):
    if n<0:
        return "Not Possible"
    elif n<=1:
        return 1
    return n*fact(n-1)
3. Palindrome Check
Check if a string is a palindrome using two-pointer approach.
def pali(s):
    l,r=0,len(s)-1
    while l<r:
        if s[l]!=s[r]:
            return False
        l+=1
        r-=1
    return True
4. Armstrong Number
Check if a number is an Armstrong number (sum of digits raised to the power of number of digits equals the number).
def arm(n):
    l=len(str(n))
    s=0
    t=n
    while t:
        d=t%10
        s+=d**l
        t//=10
    if s!=n:
        return False
    return True
5. Reverse a String
Reverse a string by prepending each character.
def revstr(s):
    r=""
    for i in s:
        r=i+r
    return r
6. Reverse a Number
Reverse digits of a number using modulo and integer division.
def revnum(n):
    r=0
    t=n
    while t:
        d=t%10
        r=r*10+d
        t//=10
    return r
7. Prime Number Check
Check if a number is prime by testing divisibility up to n//2.
def isprime(n):
    for i in range(2,n//2):
        if n%i==0:
            return False
    return True
8. Strong Number Check
A strong number is one where the sum of factorials of its digits equals the number itself. E.g., 145 = 1! + 4! + 5!
def isstrong(n):
    s=0
    t=n
    while t:
        d=t%10
        s+=fact(d)
        t//=10
    return n==s

String Problems

10. Count Characters in a String
Count frequency of each character using a dictionary.
def countchar(s):
    d={}
    for i in s:
        if i in d:
            d[i]+=1
        else:
            d[i]=1
    for i in d:
        print(i,":",d[i])
11. Remove Duplicates from String
Keep only the first occurrence of each character.
def remdup(s):
    r=""
    for i in s:
        if i not in r:
            r+=i
    return r
12. Reverse Words in Sentence (v1 — reverse order + reverse each word)
Reverse the order of words AND reverse each word individually.
def revwrdinssen1(s):
    l=s.split(" ")
    l=l[::-1]
    s=""
    for i in l:
        s+=i[::-1]+" "
    return s
13. Reverse Words in Sentence (v2 — reverse each word only)
Reverse each word individually but keep the original order.
def revwrdinssen2(s):
    l=s.split(" ")
    s=""
    for i in l:
        s+=i[::-1]+" "
    return s

Pattern Problems

14. Triangular Patterns (Star & Number)
Star pattern:   * / ** / *** / ****
Number pattern: 1 / 12 / 123 / 1234 
# Star Pattern
def star_triangle(n):
    for i in range(1, n+1):
        print("*" * i)

# Number Pattern
def num_triangle(n):
    for i in range(1, n+1):
        for j in range(1, i+1):
            print(j, end="")
        print()

Matrix Operations

15. Matrix Addition — O(r×c)
Add two matrices element by element.
def add_matrix(a, b, r, c):
    result = [[0]*c for _ in range(r)]
    for i in range(r):
        for j in range(c):
            result[i][j] = a[i][j] + b[i][j]
    return result
16. Matrix Transpose — O(r×c)
Swap rows and columns of a matrix.
def transpose(a, r, c):
    result = [[0]*r for _ in range(c)]
    for i in range(r):
        for j in range(c):
            result[j][i] = a[i][j]
    return result
17. Matrix Multiplication — O(n³)
Multiply two matrices. Condition: columns of A must equal rows of B.
def multiply(a, b, r1, c1, r2, c2):
    if c1 != r2:
        return "Cannot multiply! c1 must equal r2"
    result = [[0]*c2 for _ in range(r1)]
    for i in range(r1):
        for j in range(c2):
            for k in range(c1):
                result[i][j] += a[i][k] * b[k][j]
    return result

# Example
a = [[1,2,3],[4,5,6]]       # 2×3
b = [[7,8],[9,10],[11,12]]   # 3×2
print(multiply(a, b, 2, 3, 3, 2))

OOP — Banking & ATM System

18. Bank, Savings, Current Account & ATM (Inheritance)
Bank — Base class with deposit, withdraw, check balance.
Sa (Savings) — Inherits Bank, overrides withdrawal.
Ca (Current) — Inherits Bank, allows ₹1000 overdraft.
Atm — Menu-driven ATM interface using composition. 
class Bank():
    def __init__(self, balance=0):
        self.balance=balance

    def checkbalance(self):
        print("Balance: ",self.balance)

    def withdrawl(self, amount):
        if amount<=self.balance:
            self.balance-=amount
            print(f"Withdrawl of {amount} Successfull")
        else:
            print("Insufficient funds")

    def deposite(self, amount):
        self.balance+=amount
        print("Amount Deposited Successfully")


class Sa(Bank):
    def withdrawl(self, amt):
        if amt<=self.balance:
            self.balance-=amt
            print(amt," Withdrawed from savings account")
        else:
            print("Insufficient funds in savings account")


class Ca(Bank):
    def withdrawl(self, amt):
        limit=self.balance+1000
        if amt<=limit:
            self.balance-=amt
            print(amt," successfully withdrawn from current account")
        else:
            print("Insufficient funds in your current account")


class Atm:
    def __init__(self, account):
        self.account=account

    def start(self):
        while True:
            print("1.Deposite\n2.Withdrawl\n3.GetBalance\n4.Exit")
            n=int(input())
            if n==1:
                amt=int(input("Enter the amount to deposite"))
                self.account.deposite(amt)
            elif n==2:
                amt=int(input("Enter the amount to withdraw"))
                self.account.withdrawl(amt)
            elif n==3:
                self.account.checkbalance()
            elif n==4:
                print("Thank you for using our services!")
                break
            else:
                print("Invalid Choice")
