Generate the first n Fibonacci numbers using iteration.
⏱ Time: O(n)  |  📦 Space: O(n)
💻 Code
def fibo(n):
    r=[]
    a=0
    b=1
    for i in range(n):
        r.append(a)
        a,b=b,a+b
    return r

print(fibo(8))

3. Palindrome Check
Check if a string is a palindrome using two-pointer approach.
⏱ Time: O(n)  |  📦 Space: O(1)
💻 Code
def pali(s):
    l,r=0,len(s)-1
    while l<r:
        if s[l]!=s[r]:
            return False
        l+=1
        r-=1
    return True

print(pali("madam"))
print(pali("hello"))
📤 Output
True
False
4. Armstrong Number
Sum of digits raised to the power of number of digits equals the number. E.g., 153 = 1³+5³+3³
⏱ Time: O(d) where d = number of digits  |  📦 Space: O(1)
💻 Code
def arm(n):
    l=len(str(n))
    s=0
    t=n
    while t:
        d=t%10
        s+=d**l
        t//=10
    return s==n

9. GCD & LCM
GCD using Euclidean algorithm. LCM = (a×b) / GCD(a,b).
⏱ Time: O(log(min(a,b)))  |  📦 Space: O(1)
💻 Code
def gcd(a, b):
    while b:
        a, b = b, a % b
    return a

def lcm(a, b):
    return (a * b) // gcd(a, b)

print("GCD(12,18):", gcd(12, 18))
print("LCM(12,18):", lcm(12, 18))
print("GCD(7,13):", gcd(7, 13))
print("LCM(4,6):", lcm(4, 6))
📤 Output

LCM(12,18): 36

LCM(4,6): 12

Searching Algorithms

17. Linear Search — O(n)
Search element by checking each item one by one.
⏱ Time: O(n)  |  📦 Space: O(1)
💻 Code
def linear_search(arr, target):
    for i in range(len(arr)):
        if arr[i] == target:
            return i
    return -1

18. Binary Search — O(log n)
Efficient search on sorted array. Divides search space in half each step.
⏱ Time: O(log n)  |  📦 Space: O(1)
💻 Code
def binary_search(arr, target):
    l, r = 0, len(arr) - 1
    while l <= r:
        mid = (l + r) // 2
        if arr[mid] == target:
            return mid
        elif arr[mid] < target:
            l = mid + 1
        else:
            r = mid - 1
    return -1

a = [2, 5, 8, 12, 16, 23, 38, 56, 72, 91]
print(binary_search(a, 23))
print(binary_search(a, 50))
📤 Output
5
-1

Sorting Algorithms

19. Bubble Sort — O(n²)
Repeatedly swap adjacent elements if they're in wrong order. Stable sort.
⏱ Time: O(n²) worst/avg, O(n) best  |  📦 Space: O(1)
💻 Code
def bubble_sort(arr):
    n = len(arr)
    for i in range(n - 1):
        swapped = False
        for j in range(n - 1 - i):
            if arr[j] > arr[j + 1]:
                arr[j], arr[j+1] = arr[j+1], arr[j]
                swapped = True
        if not swapped:
            break
    return arr

a = [64, 34, 25, 12, 22, 11, 90]
print(bubble_sort(a))
📤 Output
[11, 12, 22, 25, 34, 64, 90]
20. Selection Sort — O(n²)
Find the minimum element and place it at the beginning. Repeat for unsorted portion.
⏱ Time: O(n²) always  |  📦 Space: O(1)
💻 Code
def selection_sort(arr):
    n = len(arr)
    for i in range(n):
        min_idx = i
        for j in range(i + 1, n):
            if arr[j] < arr[min_idx]:
                min_idx = j
        arr[i], arr[min_idx] = arr[min_idx], arr[i]
    return arr

a = [64, 25, 12, 22, 11]
print(selection_sort(a))
📤 Output
[11, 12, 22, 25, 64]
21. Insertion Sort — O(n²)
Build sorted array one element at a time by inserting each into correct position. Best for nearly sorted data.
⏱ Time: O(n²) worst, O(n) best  |  📦 Space: O(1)
💻 Code
def insertion_sort(arr):
    for i in range(1, len(arr)):
        key = arr[i]
        j = i - 1
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
        arr[j + 1] = key
    return arr

a = [12, 11, 13, 5, 6]
print(insertion_sort(a))
📤 Output
[5, 6, 11, 12, 13]
22. Quick Sort — O(n log n) avg
Divide and conquer: pick pivot, partition array, recursively sort sub-arrays.
⏱ Time: O(n log n) avg, O(n²) worst  |  📦 Space: O(log n)
💻 Code
def quick_sort(arr):
    if len(arr) <= 1:
        return arr
    pivot = arr[-1]
    left = [x for x in arr[:-1] if x <= pivot]
    right = [x for x in arr[:-1] if x > pivot]
    return quick_sort(left) + [pivot] + quick_sort(right)

a = [10, 7, 8, 9, 1, 5]
print(quick_sort(a))
📤 Output
[1, 5, 7, 8, 9, 10]
23. Merge Sort — O(n log n)
Divide array in half, recursively sort, then merge. Stable sort. Consistent O(n log n).
⏱ Time: O(n log n) always  |  📦 Space: O(n)
💻 Code
def merge_sort(arr):
    if len(arr) <= 1:
        return arr
    mid = len(arr) // 2
    left = merge_sort(arr[:mid])
    right = merge_sort(arr[mid:])
    return merge(left, right)

def merge(l, r):
    result = []
    i = j = 0
    while i < len(l) and j < len(r):
        if l[i] <= r[j]:
            result.append(l[i])
            i += 1
        else:
            result.append(r[j])
            j += 1
    result.extend(l[i:])
    result.extend(r[j:])
    return result

a = [38, 27, 43, 3, 9, 82, 10]
print(merge_sort(a))
📤 Output
[3, 9, 10, 27, 38, 43, 82]

Matrix Operations

24. Matrix Addition — O(r×c)
Add two matrices element by element.
💻 Code
def add_matrix(a, b, r, c):
    result = [[0]*c for _ in range(r)]
    for i in range(r):
        for j in range(c):
            result[i][j] = a[i][j] + b[i][j]
    return result

a = [[1,2],[3,4]]
b = [[5,6],[7,8]]
print(add_matrix(a, b, 2, 2))
📤 Output
[[6, 8], [10, 12]]
25. Matrix Transpose — O(r×c)
Swap rows and columns.
💻 Code
def transpose(a, r, c):
    result = [[0]*r for _ in range(c)]
    for i in range(r):
        for j in range(c):
            result[j][i] = a[i][j]
    return result

a = [[1,2,3],[4,5,6]]
print(transpose(a, 2, 3))
📤 Output
[[1, 4], [2, 5], [3, 6]]
26. Matrix Multiplication — O(n³)
Multiply two matrices. Columns of A must equal rows of B.
💻 Code
def multiply(a, b, r1, c1, r2, c2):
    if c1 != r2:
        return "Cannot multiply!"
    result = [[0]*c2 for _ in range(r1)]
    for i in range(r1):
        for j in range(c2):
            for k in range(c1):
                result[i][j] += a[i][k] * b[k][j]
    return result

a = [[1,2,3],[4,5,6]]
b = [[7,8],[9,10],[11,12]]
print(multiply(a, b, 2, 3, 3, 2))
📤 Output
[[58, 64], [139, 154]]

Linked List

27. Singly Linked List — All Operations
Operations: Insert at beginning/end/position, Delete at beginning/end/position, Display, Reverse, Detect Cycle, Find Middle.
Time: Insert/Delete at beginning O(1), at end O(n), at position O(k), Reverse O(n), Cycle O(n), Middle O(n).
💻 Code
class Node:
    def __init__(self, data):
        self.data = data
        self.next = None

class LL:
    def __init__(self, val):
        self.head = Node(val)

    def insatbeg(self, data):
        t = Node(data)
        t.next = self.head
        self.head = t

    def insatend(self, data):
        t = Node(data)
        temp = self.head
        while temp.next:
            temp = temp.next
        temp.next = t

    def insatpos(self, k, data):
        t = Node(data)
        temp = self.head
        i = 0
        while i < (k - 1):
            temp = temp.next
            i += 1
        t.next = temp.next
        temp.next = t

    def delatbeg(self):
        if self.head:
            self.head = self.head.next
        else:
            print("No elements to delete")

    def delatend(self):
        if not self.head:
            return "No"
        if self.head.next == None:
            self.head = None
            return
        temp = self.head
        while temp.next.next:
            temp = temp.next
        temp.next = None

    def delatpos(self, k):
        if not self.head:
            return "No"
        if k == 0:
            self.delatbeg()
            return
        i = 0
        temp = self.head
        while i < (k - 1):
            temp = temp.next
            i += 1
        temp.next = temp.next.next

    def show(self):
        temp = self.head
        while temp.next:
            print(temp.data, end="->")
            temp = temp.next
        print(temp.data)

    def rev(self):
        temp = self.head
        prev = None
        while temp:
            nxt = temp.next
            temp.next = prev
            prev = temp
            temp = nxt
        self.head = prev

    def has_cycle(self):
        slow = self.head
        fast = self.head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
            if slow == fast:
                return True
        return False

    def middle(self):
        slow = self.head
        fast = self.head
        while fast and fast.next:
            slow = slow.next
            fast = fast.next.next
        return slow.data

# --- Driver ---

l.insatbeg(2)
lis = [1, 2, 3, 4, 5]
for i in lis:

l.show()

OOP — Banking & ATM System

28. Bank, Savings, Current Account & ATM (Inheritance)
Bank — Base class with deposit, withdraw, check balance.
Sa (Savings) — Inherits Bank, overrides withdrawal.
Ca (Current) — Inherits Bank, allows ₹1000 overdraft.
Atm — Menu-driven ATM interface using composition.
💻 Code
class Bank():
    def __init__(self, balance=0):
        self.balance = balance

    def checkbalance(self):
        print("Balance:", self.balance)

    def withdrawl(self, amount):
        if amount <= self.balance:
            self.balance -= amount
            print(f"Withdrawal of {amount} Successful")
        else:
            print("Insufficient funds")

    def deposite(self, amount):
        self.balance += amount
        print("Amount Deposited Successfully")


class Sa(Bank):
    def withdrawl(self, amt):
        if amt <= self.balance:
            self.balance -= amt
            print(amt, "Withdrawn from savings")
        else:
            print("Insufficient funds in savings")


class Ca(Bank):
    def withdrawl(self, amt):
        limit = self.balance + 1000
        if amt <= limit:
            self.balance -= amt
            print(amt, "Withdrawn from current")
        else:
            print("Insufficient funds in current")


class Atm:
    def __init__(self, account):
        self.account = account

    def start(self):
        while True:
            print("\n1.Deposit 2.Withdraw"
                  " 3.Balance 4.Exit")
            n = int(input("Choice: "))
            if n == 1:
                amt = int(input("Amount: "))
                self.account.deposite(amt)
            elif n == 2:
                amt = int(input("Amount: "))
                self.account.withdrawl(amt)
            elif n == 3:
                self.account.checkbalance()
            elif n == 4:
                print("Thank you!")
                break
            else:
                print("Invalid Choice")
